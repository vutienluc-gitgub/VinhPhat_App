📋 Hướng dẫn tạo Google Apps Script
  1. Mở Google Sheets → menu Extensions → Apps Script
  2. Xóa code cũ, dán toàn bộ code Apps Script bên dưới vào
  3. Nhấn Save (Ctrl+S), đặt tên project bất kỳ
  4. Nhấn Deploy → New deployment
  5. Chọn type = Web app, Execute as = Me, Who has access = Anyone
  6. Nhấn Deploy → copy URL → dán vào ô trên
  7. Lần đầu sẽ hỏi quyền — nhấn Review permissions → Allow

const SHEET_NAMES = {
  nhapVaiMoc:   'Nhập vải mộc',
  vaiThanhPham: 'Vải thành phẩm',
  phieuXuatKho: 'Phiếu xuất kho',
  noKhach:      'Nợ khách',
  danhMucKH:    'Danh mục khách hàng',
};

// ── Entry points ──────────────────────────────────────────────

function doGet(e) {
  return json({ ok: true, msg: 'Vĩnh Phát API sẵn sàng ✓' });
}

function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.tryLock(15000);
  try {
    const data = JSON.parse(e.postData.contents);
    const ss   = SpreadsheetApp.getActiveSpreadsheet();
    let result;
    if      (data.action === 'nhapVaiMoc')   result = handleNhapVaiMoc(ss, data);
    else if (data.action === 'vaiThanhPham') result = handleVaiTP(ss, data);
    else if (data.action === 'xuatKho')      result = handleXuatKho(ss, data);
    else if (data.action === 'thuTien')      result = handleThuTien(ss, data);
    else if (data.action === 'themKH')       result = handleThemKH(ss, data);
    else                                     result = { ok: false, msg: 'Action không hợp lệ: ' + data.action };
    return json(result);
  } catch (err) {
    return json({ ok: false, msg: err.message });
  } finally {
    lock.releaseLock();
  }
}

// ── Helper: trả JSON response ─────────────────────────────────

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── Helper: lấy số thứ tự tiếp theo từ cột ID ────────────────

function nextSeq(ws) {
  var last = ws.getLastRow();
  if (last < 2) return 1;
  var vals = ws.getRange(2, 1, last - 1, 1).getValues();
  var max  = 0;
  for (var i = 0; i < vals.length; i++) {
    var v = String(vals[i][0]);
    var m = v.match(/(\d+)$/);
    if (m) {
      var n = parseInt(m[1], 10);
      if (n > max) max = n;
    }
  }
  return max + 1;
}

function pad(n, len) {
  var s = String(n);
  while (s.length < len) s = '0' + s;
  return s;
}

// ── Handler: Nhập vải mộc ─────────────────────────────────────
// Ghi 1 dòng lô + các cây kg vào cột C-1, C-2...
// Columns: ID | Ngày | Nhà dệt | Nhà nhuộm | Tên SP | Máy dệt |
//          TL mộc | Xe | Nợ dệt | Giá dệt | Tổng cân | Tổng cây |
//          C-1 | C-2 | ... C-N

function handleNhapVaiMoc(ss, d) {
  var ws  = ss.getSheetByName(SHEET_NAMES.nhapVaiMoc);
  var seq = nextSeq(ws);
  var yr  = new Date().getFullYear();
  var lotId = 'NVM-' + yr + '-' + pad(seq, 3);

  var kgs      = d.kgs || [];
  var tongCan  = 0;
  for (var i = 0; i < kgs.length; i++) tongCan += parseFloat(kgs[i]) || 0;
  tongCan = Math.round(tongCan * 100) / 100;

  var giaDet  = parseFloat(d.giaDet) || 0;
  var noDet   = Math.round(giaDet * tongCan);
  var tongCay = kgs.length;

  // Build row: fixed cols (12) + kg per roll
  var row = [
    lotId,
    d.ngay,
    d.nhaDet   || '',
    d.nhaNhuom || '',
    d.tenHang  || '',
    d.loaiMay  || '',
    d.tlMoc    || '',
    d.xe       || '',
    noDet,
    giaDet,
    tongCan,
    tongCay
  ];
  for (var j = 0; j < kgs.length; j++) {
    row.push(parseFloat(kgs[j]) || 0);
  }

  ws.appendRow(row);

  return {
    ok:      true,
    lotId:   lotId,
    tongCay: tongCay,
    tongCan: tongCan,
    noDet:   noDet
  };
}

// ── Handler: Thu tiền ─────────────────────────────────────────
// Tìm dòng khách hàng trong sheet Nợ khách, cộng thêm số tiền
// vào cột "Đã thanh toán" (cột H = index 7, 0-based)

function handleThuTien(ss, d) {
  var ws   = ss.getSheetByName(SHEET_NAMES.noKhach);
  var last = ws.getLastRow();

  if (last < 3) {
    return { ok: false, msg: 'Sheet Nợ khách chưa có dữ liệu' };
  }

  // Đọc toàn bộ dữ liệu (bỏ qua row 1 = banner, row 2 = header)
  var allData = ws.getRange(1, 1, last, ws.getLastColumn()).getValues();

  // Tìm header row — dòng có "TÊN KHÁCH HÀNG"
  var headerRow = -1;
  for (var i = 0; i < allData.length; i++) {
    for (var j = 0; j < allData[i].length; j++) {
      if (String(allData[i][j]).indexOf('KHÁCH HÀNG') !== -1 ||
          String(allData[i][j]).indexOf('Tên khách') !== -1) {
        headerRow = i;
        break;
      }
    }
    if (headerRow !== -1) break;
  }

  // Tìm cột "Đã thanh toán"
  var daTTCol = -1;
  var tenKHCol = -1;
  if (headerRow !== -1) {
    for (var k = 0; k < allData[headerRow].length; k++) {
      var h = String(allData[headerRow][k]).toLowerCase();
      if (h.indexOf('thanh toán') !== -1 || h.indexOf('đã tt') !== -1) daTTCol = k;
      if (h.indexOf('tên khách') !== -1 || h.indexOf('khách hàng') !== -1) tenKHCol = k;
    }
  }

  // Fallback: dùng vị trí cố định theo cấu trúc file gốc
  // Sheet Nợ khách: cột C (index 2) = Tên KH, cột H (index 7) = Đã thanh toán
  if (tenKHCol === -1) tenKHCol = 2;
  if (daTTCol  === -1) daTTCol  = 7;

  // Tìm dòng khách hàng (bắt đầu từ row sau header)
  var startRow = (headerRow !== -1) ? headerRow + 2 : 3;
  var foundRow = -1;

  for (var r = startRow - 1; r < allData.length; r++) {
    var tenKH = String(allData[r][tenKHCol] || '').trim();
    if (tenKH === d.khachHang.trim()) {
      foundRow = r + 1; // 1-based row number cho Sheets API
      break;
    }
  }

  if (foundRow === -1) {
    // KH không có trong Nợ khách → ghi dòng mới
    var newRow = [];
    for (var x = 0; x < 10; x++) newRow.push('');
    newRow[tenKHCol] = d.khachHang;
    newRow[daTTCol]  = parseFloat(d.soTien) || 0;
    ws.appendRow(newRow);
    return {
      ok:       true,
      khachHang: d.khachHang,
      daTT:      parseFloat(d.soTien) || 0,
      note:      'Tạo dòng mới vì KH chưa có trong Nợ khách'
    };
  }

  // Cộng thêm vào giá trị hiện tại
  var cell   = ws.getRange(foundRow, daTTCol + 1); // +1 vì getRange dùng 1-based col
  var oldVal = parseFloat(cell.getValue()) || 0;
  var newVal = oldVal + (parseFloat(d.soTien) || 0);
  cell.setValue(newVal);

  return {
    ok:        true,
    khachHang: d.khachHang,
    soTienThu: parseFloat(d.soTien) || 0,
    daTT:      newVal,
    ghiChu:    d.ghiChu || ''
  };
}

// ── Handler: Thêm khách hàng ──────────────────────────────────
// Columns: id-customer | Ngày tháng | Tên KH | Người phụ trách |
//          S1 | S2 | S3

function handleThemKH(ss, d) {
  var ws  = ss.getSheetByName(SHEET_NAMES.danhMucKH);
  var seq = nextSeq(ws);
  var id  = 'KH-' + pad(seq, 3);

  ws.appendRow([
    id,
    d.ngay      || new Date().toLocaleDateString('vi-VN'),
    d.ten       || '',
    d.phuTrach  || 'Vũ Tiến Lực',
    '',
    '',
    ''
  ]);

  return {
    ok:  true,
    id:  id,
    ten: d.ten
  };
}
